public class ImplicitDataTypeConversion {
	
	public static void main(String args[])
	{
		int a = 100;
		System.out.println("Int represenatation : " + a);
		
		long b = a;
		System.out.println("long represenatation : " + b);
		
		float c = b;
		System.out.println("float represenatation : " + c);


		
	}

}
