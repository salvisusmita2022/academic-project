public class ExplicitDataTypeConversion {
	
	public static void main(String args[])
	{
		double a = 50.50;
		System.out.println("Double representation is: " + a);
		
		float f = (float)a;
		System.out.println("Float representation is: " + f);
		
		long b = (long)a;
		System.out.println("long representation is: " + b);
		
		int c = (int)b;
		System.out.println("Int representation is: " + c);

		
		


		
		
	}

}
